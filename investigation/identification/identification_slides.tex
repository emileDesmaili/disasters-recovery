\section{The Problem: Recovery Time Tracks Cyclone Frequency}
% ===============================================================

{\hideframenumber
\begin{frame}
\sectionpage
\end{frame}
\showframenumber}

\begin{frame}{The Question}

\textbf{Starting point:} cyclones reduce GDP --- standard local projection confirms this.

\smallskip
\textbf{Question:} does the damage depend on \emph{how recently} the country was last struck?

\smallskip
\begin{tabular}{@{}p{4.6cm}p{6.8cm}@{}}
\toprule
Story & Prediction \\
\midrule
\textbf{Compounding vulnerability} \newline
{\small fiscal space exhausted, infrastructure weakened}
  & Bigger GDP hit when struck \emph{again soon} \\[2pt]
\textbf{Adaptation / asset depletion} \newline
{\small less capital to destroy, more prepared communities}
  & Smaller GDP hit when struck again soon \\
\bottomrule
\end{tabular}

\smallskip
\textbf{Short answer:} yes --- recently-struck countries suffer larger losses.
But the effect is concentrated in Asia.

\smallskip
\textbf{This deck:} (1) address the identification challenge;
(2) show the main finding; (3) document the geographic concentration.
\end{frame}

\begin{frame}{Why Recovery Time Is Not Randomly Assigned}

\begin{itemize}
  \item Countries struck more often have shorter gaps \emph{by construction}.
        A country hit 3×/decade cannot have a 10-year gap.
  \item $\text{cor}(\bar{W}_i,\, \tau_i) \approx -0.75$:
        high-frequency countries almost never appear in the ``isolated'' group.
  \item The raw LP gradient may just reflect \emph{which countries} are isolated
        --- not whether recovery time causes different damage.
  \item In the most cyclone-prone countries, 98.5\,\% of observations have
        $\tau \leq 2$ years $\Rightarrow$ LP must extrapolate to estimate
        the ``long recovery'' counterfactual.
\end{itemize}
\end{frame}

\section{Solution: Country-Specific Shock Thresholds}
% ===============================================================

{\hideframenumber
\begin{frame}
\sectionpage
\end{frame}
\showframenumber}

\begin{frame}{Fix: Define ``Major'' Relative to Each Country's Own History}

\textbf{Country-specific threshold:} a major cyclone is one above the
\emph{country's own} historical 95th percentile.
\[
  \text{shock}^{95}_{it} = \mathbf{1}\!\left[w_{it} \geq \hat{q}^{95}_i\right],
  \qquad
  \tau^{95}_{it} = \text{years since last such event}
\]

\medskip
\begin{itemize}
  \item Even the most cyclone-prone country qualifies $\approx 5$\,\% of years.
  \item Long recovery gaps exist in \emph{every} country --- no more extrapolation.
  \item The frequency--gap confound falls from $-0.75$ to $-0.18$.
\end{itemize}
\end{frame}

\begin{frame}{Country-Specific Thresholds Substantially Reduce the Confound}

\begin{center}
\begin{tabular}{@{}lrp{4.2cm}@{}}
\toprule
Shock definition & cor(freq., gap) & Gap variation \\
\midrule
Any strike (absolute) & $-0.75$ & Mostly short in high-freq.\ countries \\
\midrule
Country p75 & $-0.42$ & Better \\
Country p90 & $-0.32$ & Better \\
Country \textbf{p95} & $\mathbf{-0.18}$ & \textbf{Substantially better} \\
Country p99 & $+0.01$ & Near-zero confound \\
\bottomrule
\end{tabular}
\end{center}

\medskip\small
At p95 the confound drops from $-0.75$ to $-0.18$.
Long recovery gaps now exist in \emph{all} countries.
We proceed with p95 as the baseline.
\end{frame}

% ===============================================================
\section{Takeaways}
% ===============================================================

{\hideframenumber
\begin{frame}
\sectionpage
\end{frame}
\showframenumber}

\begin{frame}{Summary}
\small\vspace{-2pt}
\begin{enumerate}
  \item \textbf{Raw LP is confounded.} Recovery gap $\tau$ tracks cyclone frequency
        --- high-frequency countries almost never have long gaps.
        Country-specific p95 thresholds reduce the confound from $-0.75$ to $-0.18$.

  \item \textbf{A gradient survives.} Recently-struck countries lose \textbf{$\approx$2.6\,pp more
        GDP} at year~5 than after a 20-year recovery window ($p = 0.069$,
        consistent across all gap lengths).

  \item \textbf{Driven by Asia.} Dropping Asia collapses the gradient; no
        other continent does. The compounding effect is concentrated in
        East/South-East Asia.

  \item \textbf{Open question.} Global average masks heterogeneity.
        Whether the channel is fiscal space, capital stock, or institutional
        capacity remains to be tested.

  \item \textbf{Next step.} IV using basin-wide TC activity as a shift-share
        instrument for recovery gap length.
\end{enumerate}
\normalsize
\end{frame}

% ── Appendix: Identification of the wind shock ─────────────────

\begin{frame}{Maxwind Is Approximately Quasi-Random Within Country}

\textbf{Claim:} conditional on a country's long-run cyclone exposure,
the realised wind speed in any given year is approximately exogenous.

\begin{itemize}
  \item TC tracks and intensity are governed by sea-surface temperatures,
        wind shear, and stochastic convective dynamics --- not by GDP.
  \item Near-zero autocorrelation of the strike indicator at all lags
        $\Rightarrow$ strikes are approximately i.i.d.\ within country.
  \item ENSO controls do not attenuate the estimated GDP effect
        (shown below).
\end{itemize}

\medskip
\textbf{What this supports:} a within-country LP estimating the GDP
response to an unexpected wind shock.

\medskip
\textbf{What this does \emph{not} support:} identification of how the
response depends on \emph{when} the country was last struck --- that
requires the additional argument above.
\end{frame}

% ── Appendix: W̄-subsample approach (failed alternative) ───────

\begin{frame}{Alternative Approach: Restrict to Similar Climatologies}

\textbf{Idea:} compare high-$\tau$ and low-$\tau$ events \emph{within}
a narrow band of cyclone-prone countries.
If the gradient is a frequency confound, it should vanish inside a
homogeneous-frequency subsample.

\medskip
\textbf{Problem:} within the most cyclone-prone countries,
98.5\,\% of all observations have $\tau \leq 2$ years.
There is essentially no ``isolated event'' to compare against ---
the LP must extrapolate.

\medskip
\textbf{Result:} the apparent gradient \emph{grows} as we restrict to
higher-frequency countries, but this is an artefact of extrapolation,
not a genuine treatment contrast.
\end{frame}

% ── Appendix: Technical details ────────────────────────────────

\begin{frame}{Technical: The Compound LP Model}
\small For each horizon $h = 0, \ldots, 10$, estimate by OLS:
\[
  \log\text{GDP}_{i,t+h} - \log\text{GDP}_{i,t-1}
  = \beta_1^{(h)}\, w_{it}
  + \beta_2^{(h)} (w_{it} \times \tau_{it})
  + \boldsymbol{\gamma}^{(h)\prime} X_{it}
  + \alpha_i^{(h)}(t) + \delta_{r(i)t}^{(h)} + \varepsilon_{iht}
\]
\textbf{Marginal effect} at $\tau = c$:
$\widehat{\text{ME}}(c) = \hat\beta_1^{(h)} + \hat\beta_2^{(h)}\cdot c$, delta-method SE.

\medskip
\begin{tabular}{@{}lp{8.0cm}@{}}
\toprule
Symbol & Definition \\
\midrule
$w_{it}$ & Shock: $\mathbf{1}[w_{it}\geq\hat{q}^{95}_i]$ (country-specific p95) \\
$\tau_{it}$ & Years since last country-specific p95 event \\
$X_{it}$ & Lags 1--2 of shock and GDP growth \\
$\alpha_i^{(h)}(t)$ & Country quadratic trend \\
$\delta_{r(i)t}^{(h)}$ & Region $\times$ year fixed effect \\
\bottomrule
\end{tabular}

\medskip
\textbf{SE:} Driscoll--Kraay (arbitrary serial and cross-sectional correlation).
\normalsize
\end{frame}

\begin{frame}{Technical: Smooth Parametric LP}
\small
Stack all horizons $h=0,\ldots,10$ into a single dataset.
Parameterise the IRF as a quadratic polynomial in $h$:
\[
  \text{ME}(h,\tau=g) = \beta(h) + g\cdot\gamma(h),
  \quad \beta(h) = b_0 + b_1 h + b_2 h^2, \quad
        \gamma(h) = c_0 + c_1 h + c_2 h^2
\]
6 parameters instead of $2\times11=22$, improving precision.

\medskip
\textbf{FE:} country $\times$ horizon factor (trend) and year $\times$ horizon factor.
\textbf{SE:} clustered by country.
\normalsize
\end{frame}

\begin{frame}{IV Sketch: A Possible Causal Test}
\small
\textbf{Instrument:} basin-wide TC count in year $t-1$ (excluding own country)
$\times$ country's historical exposure share.
\[
  Z_{it} = \underbrace{\bar{e}_i}_{\text{fixed share}} \times
           \underbrace{N^{\text{basin}}_{b(i),t-1}}_{\text{annual shift}}
\]
\textbf{Why exclusion is plausible:}
basin-wide TC activity is driven by hemispheric SSTs and ITCZ dynamics,
not by any country's economic conditions; the share is predetermined.

\medskip
\textbf{Remaining concerns:}
SSTs affect both TC activity and agriculture; the share may proxy for geography.

\medskip
\textit{The p95 analysis already provides evidence for a genuine compounding effect.
IV would provide a fully causal test and sharpen the geographic heterogeneity.}
\normalsize
\end{frame}
