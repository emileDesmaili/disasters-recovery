\section{Data and Measurement}
% ===============================================================

{\hideframenumber
\begin{frame}
\sectionpage
\end{frame}
\showframenumber}

\begin{frame}{Question}
Do cyclone shocks that follow closely after a previous shock --- \textbf{compound shocks} --- cause more or less damage than \textbf{isolated shocks}?

\medskip
Two competing hypotheses:
\begin{itemize}
  \item \textbf{Compounding damage}: A second shock hits while the economy is still recovering, amplifying the GDP loss (weakened infrastructure, depleted reserves)
  \item \textbf{Diminishing marginal damage}: The first shock already destroyed vulnerable capital, so a second shock has less left to destroy; or governments pre-position response capacity after the first event
\end{itemize}
\end{frame}

\begin{frame}{Data Sources}
\begin{itemize}
  \item \textbf{GDP}: Penn World Table (PWT 10.01) --- real GDP per capita in constant 2017 USD. 182 countries, 1950--2014 (analysis uses 1970--2014).

  \medskip
  \item \textbf{Cyclones}: IBTrACS (International Best Track Archive for Climate Stewardship) --- maximum sustained wind speed (knots) for every tropical cyclone, 1970--2015. Matched to countries by geographic proximity.

  \medskip
  \item \textbf{Treatment variable}: $\mathrm{shock}_{it} = \mathbf{1}[\text{max wind}_{it} \geq 83 \text{ knots}]$ (Category~2+ on the Saffir--Simpson scale). Baseline uses Cat~2+ rather than the p95 threshold (60 knots) to avoid near-continuous treatment in large cyclone-prone countries.

  \medskip
  \item \textbf{Samples}: (i) 28 ever-treated countries (1,288 obs); (ii) 182 same-continent controls (8,372 obs). 191 Cat~2+ shock-years total.
\end{itemize}
\end{frame}

\begin{frame}{Why Cat~2+ Instead of p95?}
The p95 threshold (60 knots) is barely tropical storm strength. At p95, large countries are treated nearly every year:

\centering
\smallskip
\begin{tabular}{l r r r r}
\hline
\textbf{Country} & \textbf{p95 (60 kts)} & \textbf{Cat 1+ (64)} & \textbf{Cat 2+ (83)} & \textbf{Cat 3+ (96)} \\
\hline
Australia    & 41 / 45 yrs & 40 & 25 & 14 \\
China        & 44 / 45 yrs & 41 & 13 & 7  \\
Philippines  & 42 / 45 yrs & 39 & 30 & 17 \\
Japan        & 40 / 45 yrs & 37 & 11 & 4  \\
\hline
\end{tabular}

\smallskip
\small
\begin{itemize}
  \item At p95, 61.5\% of shocks are compound $\to$ limited within-country variation in gap
  \item At Cat 2+, compound share drops to 40\% with 191 shocks --- a better balance of power and variation
  \item Cat 2+ captures genuinely damaging storms (sustained winds $\geq$ 83 knots / 154 km/h)
\end{itemize}
\end{frame}

\smallskip
\small
At Cat~2+, even frequently-hit countries (PHL, AUS) now have meaningful gaps between shocks. Smaller countries have sparse, isolated events.
\end{frame}

\begin{frame}{Identification Strategy}
\textbf{Two approaches}, both using the continuous gap measure:

\medskip
\textbf{1.\ Standard LP} (horizon-by-horizon):
\[
  y_{i,t+h} - y_{i,t-1} = \beta_1^h \cdot \mathrm{shock}_{it} + \beta_2^h \cdot \mathrm{shock}_{it} \times \mathrm{gap}_{it} + \gamma_h' X_{it} + \alpha_i(t) + \delta_t + \varepsilon_{it}
\]

\textbf{2.\ Parametric LP} (stacked, smooth): impose $\beta(h) = \theta_0 + \theta_1 h + \theta_2 h^2$ (same for gap interaction). Stack all horizons, estimate jointly.

\medskip
\begin{itemize}
  \item Controls: 2 lags of GDP growth and shock indicator (horizon-varying)
  \item FE: country$\times$horizon trends + year$\times$horizon FE
  \item Same-continent countries (182) serve as controls for year$\times$horizon effects
  \item SEs: Driscoll--Kraay (standard LP) / clustered by country (parametric LP)
\end{itemize}
\end{frame}

\begin{frame}{Estimation: Parametric LP with Continuous Gap}
Instead of a binary compound/isolated split, we use a \textbf{continuous measure}: $\mathrm{gap}_{it}$ = years since the country's previous Cat~2+ shock.

\medskip
\textbf{Stacked local projection} across horizons $h = 0, \ldots, 10$:
\[
  y_{i,t+h} - y_{i,t-1} = \beta(h) \cdot \mathrm{shock}_{it} + \phi(h) \cdot \mathrm{shock}_{it} \times \mathrm{gap}_{it} + \gamma_h'X_{it} + \alpha_{ih}(t) + \delta_{th} + \varepsilon_{ith}
\]
where $\beta(h) = \theta_0 + \theta_1 h + \theta_2 h^2$ and $\phi(h) = \phi_0 + \phi_1 h + \phi_2 h^2$ are \textbf{quadratic polynomials in horizon}.

\medskip
\begin{itemize}
  \item $X_{it}$: two lags of GDP growth and shock indicator, interacted with horizon
  \item $\alpha_{ih}(t)$: country$\times$horizon FE with country$\times$horizon-specific linear and quadratic trends
  \item $\delta_{th}$: year$\times$horizon FE. Identified using same-continent controls (182 countries)
  \item SEs: clustered by country
  \item 6 parameters for the shock dynamics (vs.\ $2 \times 11 = 22$ in standard LP)
\end{itemize}
\end{frame}

\begin{frame}{Robustness Checks (Binary Analysis, p95 Threshold)}
\small
The binary compound/isolated analysis (detailed in appendix) yields consistent findings:
\begin{itemize}
  \item \textbf{Binary split at p95}: Isolated shocks $-3.0\%$ at $h = 5$, compound $-1.3\%$. Difference positive but never significant.
  \item \textbf{Broader definition} (shock in $t{-}1$ or $t{-}2$): pattern persists
  \item \textbf{Cumulative exposure} (0, 1, 2+ shocks in past 3 years): non-monotonic --- 2+ recent shocks cause \emph{less} damage
  \item \textbf{Drop always-compound countries} (CHN, PHL, AUS): pattern unchanged
  \item \textbf{Continuous intensity} $\times$ recent shock: gap disappears when controlling for wind speed continuously --- suggests intensity selection drives part of binary result
  \item \textbf{No country trends}: pattern reverses, confirming trends are essential
\end{itemize}

\medskip
The smooth gap approach sidesteps the binary threshold choice and recovers the same gradient continuously.
\end{frame}

\begin{frame}{Sensitivity to Treatment Threshold}
\centering
\begin{tabular}{l r r r r r}
\hline
\textbf{Threshold} & \textbf{Shocks} & \textbf{Treated} & \textbf{\%Comp} & \textbf{Isol.\ $h{=}5$} & \textbf{Comp.\ $h{=}5$} \\
\hline
p95 (60 kts)    & 423 & 40 & 61.5\% & $-3.0\%$ & $-1.3\%$ \\
Cat 1+ (64 kts) & 374 & 36 & 58.0\% & $-2.9\%$ & $-1.9\%$ \\
p97.5 (72 kts)  & 235 & 29 & 47.2\% & $-1.3\%$ & $-0.5\%$ \\
\textbf{Cat 2+ (83 kts)} & \textbf{191} & \textbf{28} & \textbf{40.3\%} & $-1.7\%$ & $-0.8\%$ \\
p99 (90 kts)    & 101 & 23 & 17.8\% & $+0.6\%$ & $+1.2\%$ \\
Cat 3+ (96 kts) & 102 & 23 & 18.6\% & $+0.7\%$ & $+1.7\%$ \\
\hline
\end{tabular}

\bigskip
\begin{itemize}
  \item Cat 2+ (bold) offers the best balance: enough shocks (191) with moderate compound share (40\%)
  \item At moderate thresholds: compound consistently less damaging
  \item At severe thresholds (p99, Cat 3+): effects attenuate due to small samples
\end{itemize}
\end{frame}

\section{Discussion}
% ===============================================================

{\hideframenumber
\begin{frame}
\sectionpage
\end{frame}
\showframenumber}

\begin{frame}{Why Might Compound Shocks Be Less Damaging?}
\begin{enumerate}
  \item \textbf{Diminishing marginal destruction}: The first shock destroys the most vulnerable capital. A second shock has less left to destroy.

  \medskip
  \item \textbf{Preparedness and learning}: Post-shock investments in early warning, building codes, and evacuation capacity reduce vulnerability to follow-up shocks.

  \medskip
  \item \textbf{Mechanical baseline effect}: For compound shocks, $y_{i,t-1}$ is already depressed, so the measured decline from the lower baseline is mechanically smaller. Country trends partially address this.

  \medskip
  \item \textbf{Intensity selection}: Frequently-hit countries may experience weaker average storms; isolated shocks tend to be rarer but more extreme. The continuous gap measure partially addresses this by operating within countries.
\end{enumerate}
\end{frame}

\begin{frame}{Summary}
\begin{enumerate}
  \item \textbf{No evidence of compounding damage}: Repeated cyclone shocks do not amplify GDP losses. If anything, compound shocks (short gaps) are \emph{less} damaging.

  \medskip
  \item \textbf{Smooth gradient}: Replacing the binary split with a continuous gap and parametric IRFs yields a clear gradient --- longer gaps since the last shock are associated with larger damage. Robust across samples and polynomial degrees.

  \medskip
  \item \textbf{Same-continent controls don't change results}: Adding 150+ never-treated countries from treated continents leaves the estimates virtually unchanged.

  \medskip
  \item \textbf{Country heterogeneity}: Country-specific parametric IRFs show the largest effects for PHL, JPN, MEX, IND, MOZ.

  \medskip
  \item \textbf{Cat~2+ is a better baseline}: The p95 threshold (60 knots) produces near-continuous treatment for large countries. Cat~2+ (83 knots, 191 shocks) provides more within-country variation and captures genuinely severe events.
\end{enumerate}
\end{frame}
